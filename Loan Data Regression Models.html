X = df_nor.copy()
X = X.drop(targets, axis=1)
Y = df_nor[dflt]

X_train, X_test, Y_train, Y_test = train_test_split(X, Y, train_size=0.8, test_size=0.2, random_state=1)

varNames = list( X_train.columns.values )
m = LogisticRegression()
m.fit(X_train,Y_train)

coef_dict = {}
coef_dict["INTERCEPT"] = m.intercept_[0]
for coef, feat in zip(m.coef_[0],varNames):
    coef_dict[feat] = coef

print('COEFFICIENTS:')
print()

for i in coef_dict :
    print( i, " = ", coef_dict[i]  )
print()

# Training Stats
print('Training Dataset:')
pred_dflt_train = m.predict( X_train )
prob_dflt_train = m.predict_proba( X_train )
prob_dflt_train = prob_dflt_train[:,1]

fpr_train, tpr_train, threshold_train = metrics.roc_curve( Y_train, prob_dflt_train )
auc_train = metrics.auc(fpr_train, tpr_train)
print("AUC=",auc_train)

accuracy_train = metrics.accuracy_score( Y_train, pred_dflt_train )
print("ACCURACY=",accuracy_train)

print()
