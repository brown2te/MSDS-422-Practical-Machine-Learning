def TF_scores(TF_model, X, Y) :
    
    probs = TF_model.predict_proba(X)

    pred_list = []

    for p in probs:

        pred_list.append( np.argmax(p) )

    preds = np.array( pred_list )

    acc_score = metrics.accuracy_score(Y, preds)

    p1 = probs[:,1]

    fpr, tpr, threshold = metrics.roc_curve(Y, p1, drop_intermediate=False)

    auc = metrics.auc(fpr,tpr)
    
    return {"Accuracy":acc_score, "TPR":tpr, "FPR":fpr, "Thres":threshold, "AUC":auc}

def TF_amt_scores(TF_model, X, Y) :
    
    pred = TF_model.predict(X)

    MEAN = Y.mean()
    
    RMSE = math.sqrt( metrics.mean_squared_error(Y, pred))
    
    return {"RMSE":RMSE, "Mean":MEAN}
